Assuming that you are experiencing txid wraparound risk (the most common one, as opposed to mxid wraparound risk), the quickest way is to manually vacuum each table with a high age:

bash
SELECT c.oid::regclass AS table_name,
age(c.relfrozenxid) AS age,
pg_size_pretty(pg_total_relation_size(c.oid)) AS table_total_size
FROM pg_class c
WHERE c.relkind IN ('r', 'm', 't')
AND age(c.relfrozenxid) > current_setting('autovacuum_freeze_max_age')::bigint
ORDER BY 2 DESC
LIMIT 10;

Total connection details: (idle + non-idle)
++++++++++++++++++++++++++++++++++++++++++++

select 
    A.total_connections, 
    A.non_idle_connections, 
    B.max_connections,
    round((100 * A.total_connections::numeric / B.max_connections::numeric), 2) connections_utilization_pctg
from
  (select count(1) as total_connections, sum(case when state!='idle' then 1 else 0 end) as non_idle_connections from pg_stat_activity) A,
  (select setting as max_connections from pg_settings where name='max_connections') B;

frequent running queries:
+++++++++++++++++++++++++

with
a as (select dbid, queryid, query, calls s from pg_stat_statements),
b as (select dbid, queryid, query, calls s from pg_stat_statements, pg_sleep(1))
select
        pd.datname as db_name, 
        substr(a.query, 1, 400) as the_query, 
        sum(b.s-a.s) as runs_per_second
from a, b, pg_database pd
where 
  a.dbid= b.dbid 
and 
  a.queryid = b.queryid 
and 
  pd.oid=a.dbid
group by 1, 2
order by 3 desc;
