TerraForm Learning
==========================
STEP 1 — Install Terraform on Linux VM
sudo yum install -y yum-utils
sudo yum install -y wget unzip
Download Terraform:
wget https://releases.hashicorp.com/terraform/1.8.5/terraform_1.8.5_linux_amd64.zip
Unzip
unzip terraform_1.8.5_linux_amd64.zip
sudo mv terraform /usr/local/bin/
terraform version

STEP 2 — Install OCI CLI : https://docs.oracle.com/en-us/iaas/Content/API/SDKDocs/cliinstall.htm?utm_source=chatgpt.com
Quick install :
bash -c "$(curl -L https://raw.githubusercontent.com/oracle/oci-cli/master/scripts/install/install.sh)"
Oci cli installation from oracle docs: 
dnf -y install oraclelinux-developer-release-el9




Accept defaults.
Verify :
oci --version

STEP 3 — Create OCI API Key
on linux VM :
mkdir -p ~/.oci
cd ~/.oci

Generate key : 
openssl genrsa -out oci_api_key.pem 2048

Generate public key:
openssl rsa -pubout -in oci_api_key.pem -out oci_api_key_public.pem

STEP 4 — Upload Public Key to OCI
OCI Console

Then:

Profile icon (top right)
My Profile
API Keys
Add API Key
Upload public key

Upload:

~/.oci/oci_api_key_public.pem

OCI will show:

User OCID
Fingerprint
Tenancy OCID

Copy these.

vi ~/.oci/config

STEP 9 — Create main.tf

IMPORTANT:

Free Tier supports:

VM.Standard.E2.1.Micro

Use this exact code.

resource "oci_core_vcn" "my_vcn" {

  cidr_block     = "10.0.0.0/16"
  compartment_id = var.compartment_ocid
  display_name   = "terraform-vcn"
}

resource "oci_core_internet_gateway" "igw" {

  compartment_id = var.compartment_ocid
  display_name   = "terraform-igw"
  vcn_id         = oci_core_vcn.my_vcn.id
}

resource "oci_core_route_table" "rt" {

  compartment_id = var.compartment_ocid
  vcn_id         = oci_core_vcn.my_vcn.id

  route_rules {
    destination       = "0.0.0.0/0"
    destination_type  = "CIDR_BLOCK"
    network_entity_id = oci_core_internet_gateway.igw.id
  }
}

resource "oci_core_security_list" "sl" {

  compartment_id = var.compartment_ocid
  vcn_id         = oci_core_vcn.my_vcn.id

  ingress_security_rules {
    protocol = "6"

    source = "0.0.0.0/0"

    tcp_options {
      min = 22
      max = 22
    }
  }

  ingress_security_rules {
    protocol = "6"

    source = "0.0.0.0/0"

    tcp_options {
      min = 80
      max = 80
    }
  }

  egress_security_rules {
    protocol    = "all"
    destination = "0.0.0.0/0"
  }
}

resource "oci_core_subnet" "my_subnet" {

  cidr_block        = "10.0.1.0/24"
  compartment_id    = var.compartment_ocid
  vcn_id            = oci_core_vcn.my_vcn.id
  route_table_id    = oci_core_route_table.rt.id
  security_list_ids = [oci_core_security_list.sl.id]

  display_name = "terraform-subnet"

  prohibit_public_ip_on_vnic = false
}

resource "oci_core_instance" "my_instance" {

  availability_domain = data.oci_identity_availability_domains.ads.availability_domains[0].name

  compartment_id = var.compartment_ocid

  display_name = "terraform-instance"

  shape = "VM.Standard.E2.1.Micro"

  create_vnic_details {
    subnet_id        = oci_core_subnet.my_subnet.id
    assign_public_ip = true
  }

  source_details {

    source_type = "image"

    source_id = data.oci_core_images.oracle_linux.images[0].id
  }

  metadata = {
    ssh_authorized_keys = var.ssh_public_key
  }
}

data "oci_identity_availability_domains" "ads" {
  compartment_id = var.compartment_ocid
}

data "oci_core_images" "oracle_linux" {

  compartment_id           = var.compartment_ocid
  operating_system         = "Oracle Linux"
  operating_system_version = "8"
  shape                    = "VM.Standard.E2.1.Micro"
}


STEP 10 — Initialize Terraform

terraform init
Downloads OCI provider.

STEP 11 — Validate
terraform validate
STEP 12 — Preview Changes
terraform plan
You will see:

VCN
subnet
compute instance
etc.


STEP 13 — Create Infrastructure

terraform apply

Type
Yes
Terraform will create everything.


STEP 14 — Verify in OCI Console
Go to:

Compute → Instances
Networking → VCN

You will see resources created automatically.

ssh opc@<PUBLIC_IP>
