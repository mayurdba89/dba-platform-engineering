🔹 STEP 3: Add Pipeline Script

In Jenkins → Configure → Pipeline → paste:

pipeline {
    agent any

    environment {
        ORACLE_HOME = "/u01/app/oracle/product/19c/dbhome_2"
        PATH = "${ORACLE_HOME}/bin:${env.PATH}"
        DB_CONN = "c##jenkins/jenkins@//192.168.139.129:1521/ORAPROD"
    }

    stages {

        stage('Checkout Code') {
            steps {
                sh '''
                rm -rf db-project
                git clone /home/mayur/db-project
                '''
            }
        }

        stage('List Scripts') {
            steps {
                sh '''
                echo "Available scripts:"
                ls -lrt db-project/scripts
                '''
            }
        }

        stage('Execute SQL Scripts') {
            steps {
                sh '''
                echo "Executing DB scripts..."

                for file in db-project/scripts/*.sql
                do
                  echo "Running $file"
                  sqlplus -s $DB_CONN @$file
                done
                '''
            }
        }

        stage('Verify Data') {
            steps {
                sh '''
                sqlplus -s $DB_CONN <<EOF

                SET HEADING ON
                SELECT * FROM jenkins_demo;

                EXIT;
EOF
                '''
            }
        }
    }

    post {
        success {
            echo '✅ Deployment Successful'
        }
        failure {
            echo '❌ Deployment Failed'
        }
    }
}
